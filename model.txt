@Types
  pk        primary key not null
  str       varchar(10)
  
Customer
  customer_no       int pk
  name              str
  address           str
  phone_no          str

Products
  product_code      int pk
  manufacturer_no   int fk[Manufacturers]
  price             float
  type              str

Stock
  product_code      int fk[Products]
  shop_no           int fk[Shops]

Manufacturers
  manufacturer_no  int pk
  name             str

Sales_details
  product_code     int fk[Products]
  sales_no         int fk[Sales]
  quantity         int

Shops
  shop_no          int pk
  address          str
  name             str

Vendors
  staff_no         int pk
  name             text

Sales
  sales_no         int pk
  staff_no         int fk[Vendors]
  shop_no          int fk[Shops]
  date             timestamp
  sum_total        float
